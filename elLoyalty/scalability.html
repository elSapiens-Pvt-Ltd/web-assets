// Cache-aside pattern for rule loading func (s *RuleService) GetActiveRules(ctx context.Context, tc TenantContext) ([]BenefitRule, error) { cacheKey := fmt.Sprintf("loyalty:rules:%s", tc.OrganizationID) // L1: process-local cache if cached, ok := s.localCache.Load(cacheKey); ok { entry := cached.(*cacheEntry) if time.Since(entry.loadedAt) < 60*time.Second { return entry.rules, nil } } // L2: Redis data, err := s.redis.Get(ctx, cacheKey).Bytes() if err == nil { var rules []BenefitRule json.Unmarshal(data, &rules) s.localCache.Store(cacheKey, &cacheEntry{rules: rules, loadedAt: time.Now()}) return rules, nil } // L3: PostgreSQL (source of truth) rules, err := s.repo.ListActive(ctx, tc) if err != nil { return nil, err } // Populate caches data, _ = json.Marshal(rules) s.redis.Set(ctx, cacheKey, data, 10*time.Minute) s.localCache.Store(cacheKey, &cacheEntry{rules: rules, loadedAt: time.Now()}) return rules, nil }

Worker Design

// Background worker with graceful shutdown type ExpiryWorker struct { repo WalletRepository ledger LedgerRepository redis *redis.Client logger *slog.Logger } func (w *ExpiryWorker) Run(ctx context.Context) { ticker := time.NewTicker(1 * time.Hour) defer ticker.Stop() for { select { case <-ctx.Done(): w.logger.Info("expiry worker shutting down") return case <-ticker.C: expired, err := w.repo.ExpireCoins(ctx) if err != nil { w.logger.Error("expiry run failed", "err", err) continue } w.logger.Info("coins expired", "count", len(expired)) // Invalidate Redis cache for affected customers for _, e := range expired { w.redis.Del(ctx, fmt.Sprintf("loyalty:wallet:%s", e.CustomerID)) } } } }
Scale

Horizontal Scaling

Stateless Instances

elLoyalty instances are stateless. All shared state lives in PostgreSQL and Redis. Deploy multiple replicas behind a load balancer.

L1 Cache Invalidation

When a rule is updated, the service publishes to Redis pub/sub channel loyalty:cache:invalidate. All instances listen and clear their local cache.

Worker Leader Election

Background workers (expiry, tier recalc) use Redis-based leader election. Only one instance runs the worker at a time. If the leader fails, another takes over.

Read Replicas

Analytics and reporting queries route to PostgreSQL read replicas. Write operations go to the primary. Achieved via separate pgxpool connections.

Ledger Scaling

The ledger_entries table grows continuously. Partitioning and archival keep queries fast.

-- Range partition ledger_entries by month CREATE TABLE ledger_entries ( id UUID NOT NULL, created_at TIMESTAMPTZ NOT NULL DEFAULT NOW(), workspace_id UUID NOT NULL, -- ... other columns ) PARTITION BY RANGE (created_at); -- Create monthly partitions CREATE TABLE ledger_entries_2026_04 PARTITION OF ledger_entries FOR VALUES FROM ('2026-04-01') TO ('2026-05-01'); CREATE TABLE ledger_entries_2026_05 PARTITION OF ledger_entries FOR VALUES FROM ('2026-05-01') TO ('2026-06-01'); -- Archive partitions older than 12 months to cold storage -- Query planner automatically prunes partitions based on created_at filter
