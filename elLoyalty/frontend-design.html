11 — Frontend Design

POS checkout widget, admin dashboard, Zustand stores, component architecture, and offline-aware UI

Overview

Two Frontend Surfaces

elLoyalty has two distinct frontends: a checkout widget embedded in the elPOS terminal app, and an admin dashboard for configuring rules, campaigns, and viewing analytics.

POS Checkout Widget

Embedded in elPOS React app. Shows customer wallet, available benefits, toggle selection, earn preview. Must work offline. Uses GraphQL BFF when online, cached rules when offline.

Admin Dashboard

Standalone React app or tab within elPOS admin. Rule CRUD, campaign management, customer search, tier management, analytics. Uses GraphQL for nested data (rules with conditions, campaigns with enrollments). Online-only.

Zustand Store Design

// stores/loyaltyStore.ts import { create } from 'zustand' interface LoyaltyState { // Customer context customer: LoyaltyCustomer | null wallet: WalletSummary | null // Evaluation results availableBenefits: CandidateBenefit[] selectedBenefitIds: string[] conflicts: ConflictInfo[] earnPreview: EarnPreview | null // Applied state appliedBenefits: AppliedBenefit[] | null totalDiscount: number // paisa // Offline state isOffline: boolean cacheAge: Date | null offlineBudget: OfflineBudget | null // Actions setCustomer: (id: string) => Promise<void> evaluateCart: (cart: CartItem[]) => Promise<void> toggleBenefit: (benefitId: string) => void applyBenefits: () => Promise<ApplyResult> creditEarnings: (saleId: string, total: number) => Promise<void> reset: () => void } export const useLoyaltyStore = create<LoyaltyState>((set, get) => ({ customer: null, wallet: null, availableBenefits: [], selectedBenefitIds: [], conflicts: [], earnPreview: null, appliedBenefits: null, totalDiscount: 0, isOffline: !navigator.onLine, cacheAge: null, offlineBudget: null, evaluateCart: async (cart) => { const { customer, isOffline, offlineBudget } = get() if (!customer) return if (isOffline) { // Use cached rules + offline budget for local evaluation const result = await offlineEvaluate(customer, cart, offlineBudget) set({ availableBenefits: result.benefits, earnPreview: result.earn }) } else { // GraphQL BFF query — single round-trip for wallet + benefits + earn preview const result = await gqlClient.request(BENEFIT_PREVIEW_QUERY, { customerId: customer.id, cartItems: cart.map(toCartItemInput), }) set({ availableBenefits: result.benefitPreview.benefits, earnPreview: result.benefitPreview.earnPreview, conflicts: result.benefitPreview.conflicts }) } }, toggleBenefit: (benefitId) => { const { selectedBenefitIds, availableBenefits, conflicts } = get() // Check conflicts: if toggling ON an exclusive benefit, deselect conflicting ones const benefit = availableBenefits.find(b => b.id === benefitId) if (!benefit) return let next = [...selectedBenefitIds] if (next.includes(benefitId)) { next = next.filter(id => id !== benefitId) } else { // Remove conflicting exclusive benefits const conflicting = conflicts .filter(c => c.benefitId === benefitId) .map(c => c.conflictsWith) next = next.filter(id => !conflicting.includes(id)) next.push(benefitId) } set({ selectedBenefitIds: next }) }, // ... other actions }))
Admin

Admin Dashboard

Admin pages use the GraphQL BFF to query deeply nested data in a single request — rules with conditions, campaigns with enrollments, customers with full transaction history. This avoids the multiple REST round-trips that traditional CRUD would require.

Offline UI

Offline-Aware UI Patterns

Connectivity Indicator

Bottom bar shows "Online" (green) or "Offline · cached 5 min ago" (amber). Cache age shown so cashier knows data freshness.

Approximate Values

When offline, all earn previews show "~" prefix: "~95 coins". Tooltip explains "Exact amount calculated when online."

Budget Limits

If coin redemption is limited by offline budget, show: "150 of 500 coins available offline". Full balance shown but greyed.

Coupon Warnings

Offline coupon entry shows: "Code accepted — will be verified on sync." Yellow warning badge instead of green confirmed.

// types/loyalty.ts — mirrors backend models exactly interface CandidateBenefit { id: string type: 'loyalty_coins' | 'cashback' | 'loyalty_cash' | 'coupon' | 'campaign' label: string // "Use 150 coins → ₹150 off" discountPaisa: number // immediate discount in paisa coinsToDebit: number // coins consumed (0 for cashback) cashToDebit: number // wallet cash consumed in paisa stackable: boolean exclusionGroup: string | null ruleId: string score: number // higher = better deal for customer } interface EarnPreview { coinsToEarn: number cashbackPaisa: number isApproximate: boolean // true when offline } interface WalletSummary { coinBalance: number cashBalancePaisa: number activeCoupons: number tier: 'bronze' | 'silver' | 'gold' | 'platinum' } interface ConflictInfo { benefitId: string conflictsWith: string reason: string // "Exclusive: cannot stack with coins" } interface OfflineBudget { customerId: string coinBudget: number cashBudgetPaisa: number syncedAt: string // ISO timestamp }
